//	%[11]d		bArg1 as decimal number
//	%[12]d		bArg2 as decimal number

		0x27	"PCD3000 SiteInit: Denied"

//	%[18]s		bArg1 as field #3 string:
		0x00	"Level 1"

		0x02	"Level 3"

		0x18	"Dealer Program"
		0x19	"Alarm"
//	%[16]s		bypassed zone(s): ONLY for Close,special bitmap format
//	%[17]s		bArg2 as field #5 string:
		0	"PowerRestore"
		1-96	"User (1-96)"
		97	"Auto"
		98	"QL"
		99	"EasyArm"
		100	"Keyswitch"
		101-116	"Keyfob (1-16)"
//	%[20]s		bArg1 as Area (1-8)
//	%[21]s		bArg2 as Zone (1-96) description:
  "%d:[%.29s]",bArg2,ZoneID[(bArg2-1)*32+3]
  "%d",bArg2 (if description blank)
//	%[22]s		bArg2 as EZM# (1-22)
//	%[23]s		bArg2 as field #6 string:
		0	"TimeOut"
		1-96	"User (1-96)"
		97	"Auto"
		98	"QL"
		100	"Keyswitch"
		101-116	"Keyfob (1-16)"
//	%[24]s		bArg2 as Wireless ID:
 "%6.6lX-%u",PanelProg[3201+(bArg2-1)*5],
    (*(LPULONG)(&PanelProg[3202+(bArg2-1)*5])) & 0x00ffffff
//	%[25]s		bArg2 as Keyfob(1-16) ID:
  "%u:%6.6lX",bArg2,(*(LPULONG)(&PanelProg[3682+(bArg2-1)*5])) & 0x00ffffff
